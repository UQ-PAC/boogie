(set-option :print-success false)
(set-info :smt-lib-version 2.6)
(set-option :produce-interpolants true)
(set-logic QF_UFLIA)
; done setting options


(declare-fun u () Bool)
(declare-fun y () Int)
(declare-fun x () Int)
(push 1)
(assert (!  (and (not u) (not (=> (not u) (>= y 1)))) :named g1))
(assert (!  (and (= x 1) (= y 1)) :named g2))
(check-sat)
(get-interpolants g1 g2)
(pop 1)
(declare-fun |x'| () Int)
(declare-fun |y'| () Int)
(declare-fun |u'| () Bool)
(declare-fun t1 () Int)
(declare-fun t2 () Int)
(push 1)
(assert (! (let ( (a!1 (and u (=> u (<= (+ x y) 0))))) (or (<= y 0) a!1)) :named g1))
(assert (!  (or (and (= x 1) (= y 1)) (and (and (and (and (and (and (and (= |x'| 1) (= |y'| 1)) |u'|) |u'|) (= t1 |x'|)) (= t2 |y'|)) (= x (+ t1 t2))) (= y (+ t1 t2)))) :named g2))
(check-sat)
(get-interpolants g1 g2)
(pop 1)
